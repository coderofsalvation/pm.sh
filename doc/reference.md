if $i is "foo"

if not $j is "foo" and $x is "bar"

    if $j > $y and $j != $y or $j >= $y
      echo "foo"

        </code>
      </pre>

    <td>
      <pre>
        <code>
if [[ "$i" == "foo" ]]; then
  echo "foo" 
else
  echo "bar"
fi

if [[ ! "$j" == "foo" && "$x" == "bar" ]]; then
  if [[ "$j" == "foo" || "$j" == "xfoo" ]]; then
    if [[ "$j" -gt "$y" && "$j" -ne "$y" || "$j" -ge "$y" ]]; then
      echo "foo"
    fi
  fi
fi
        </code>
